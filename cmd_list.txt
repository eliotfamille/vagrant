vagrant init generic/ubuntu1804     # cree le unfichier vagrantfile avec ubuntu1804
vagrant up                          # cree le vm dans votre outil de vm et permet aussi de l'allumer si la vm existe deja
vagrant halt                        # eteindre le vm
vagrant ssh                         # connexion via ssh au vm cree
    Necessite que la vm soit allumer pour fonctionne
exit                                # quitter la connexion ssh

outil:
    git
    nginx
    php
    mysql-server
